void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  // This widget is the root of your application.
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Studio - Flutter',
      // Application theme data, you can set the colors for the application as
      // you want
      theme: ThemeData(
        primarySwatch: Colors.deepOrange,
      ),
      // A widget which will be started on application startup
      home: MyHomePage(title: 'My bakery'),
    );
  }
}

class MyHomePage extends StatefulWidget {
  final String title;
  const MyHomePage({Key? key, required this.title}) : super(key: key);

  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  int _counts = 0;
  String _counter = 'Cosmopolitan';
  String _temp = '';
  String link = 'https://www.thecocktaildb.com/api/json/v1/1/search.php?s=';
  String og_link = 'https://www.thecocktaildb.com/api/json/v1/1/search.php?s=';

  void _incrementCounter() async {
    link = og_link + _counter;
    final res = await http.get(Uri.parse(link));
    Map<String, dynamic> weatherData = jsonDecode(res.body);

    setState(() {
      _counts++;
      link = link + _counter;
      _temp = weatherData['drinks'][0]['strInstructions'].toString();
    });
  }

//   @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        // The title text which will be shown on the action bar
        title: Text('Cocktail Recipes'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            Text(
              'Type in a cocktail and find its recipe!',
              style: TextStyle(fontSize: 30),
            ),
            Text(
              'Try Margarita, Cosmopolitan, Old Fashioned, ...',
              style: TextStyle(fontSize: 20, color: Colors.grey),
            ),
            TextField(
              autofillHints: ['Margarita, Cosmopolitan'],
              onChanged: (text) {
                _counter = text;
              },
            ),
            Text(
              'You have used this app $_counts times! \nThe recipe: \n$_temp',
              style: TextStyle(fontSize: 20),
            )
          ],
        ),
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: _incrementCounter,
        tooltip: 'Increment',
        child: Icon(Icons.favorite),
      ),
    );
  }
}
